- [ ] **Impl** â€” `src/validate-requires.ts` (pure, no fs):
```ts
export interface MigrationLike { id: string; requires?: string[] }

/** Pack-time guard: ids unique, every `requires` resolves to a strictly earlier id. */
export function validateRequires(migrations: MigrationLike[]): void {
  const seen = new Set<string>();
  for (let i = 0; i < migrations.length; i++) {
    const { id, requires = [] } = migrations[i];
    if (seen.has(id)) throw new Error(`Duplicate migration id: ${id}`);
    for (const req of requires) {
      if (!seen.has(req)) {
        throw new Error(
          `Migration ${id} requires "${req}", which is not an earlier migration (forward ref, typo, or unknown id).`,
        );
      }
    }
    seen.add(id);
  }
}
```
